# FormGroup

A group component ready to use in your form. 

FormGroup is UI/UX agnostic so you need to provide your style to have the look and feel you prefer.

## Usage

Use the group component when users can only select one option from a list.

It provides heading, hint text and error message components. 

An example with all the available properties is:

```js
<FormGroup
    name='name-of-group'
    inputProps={{
        className: 'shared input class names'
    }}
    itemProps={{
        className: 'shared item class names'
    }}
    labelProps={{
        className: 'shared label class names'
    }}
    items={[
        {
            label: 'item-label',
            value: 'yes',
            ariaLabel: 'aria-label',
            disabled: false,
            hint: {
                text: 'hint-text',
                classes: 'class names',
                id: 'id'
            },
            id: 'id',
            inputProps: {
                style: {
                    lineHeight: '10px'
                }
            },
            itemProps: {
                style: {
                    display: 'flex'
                }
            },
            labelProps: {
                style: {
                    fontSize: '14px'
                }
            },
            selected: false,
            onChange: handleChange
        },
        {
            text: 'divider',
            className: 'divider classes',
            id: 'divider-id'
        },
        {
            label: 'item-label',
            value: 'no',
            ariaLabel: 'aria-label',
            disabled: false,
            hint: {
                text: 'hint-text',
                classes: 'class names',
                id: 'id'
            },
            id: 'id',
            inputProps: {
                style: {
                    lineHeight: '10px'
                }
            },
            itemProps: {
                style: {
                    display: 'flex'
                }
            },
            labelProps: {
                style: {
                    fontSize: '14px'
                }
            },
            selected: false,
            onChange: handleChange
        }
    ]}
    ariaDescribedBy='aria-described-by'
    ariaLabelledBy='aria-labelled-by'
    groupClassess='group-classes'
    error={{
       text: 'error message'
       classes: 'error-classes'
       id: 'error-id',
       visuallyHiddenText: {
           classes='visually-hidden-classes',
           text='visually hidden text'
       }
    }}
    fieldsetClasses='fieldset classes'
    hint={{
        text: 'hint-text',
        classes: 'class names',
        id: 'hint-id'
    }}
    id='id'
    itemsClasses='class names for the group items'
    legend={{
        text: 'heading',
        className: 'legend classes',
        heading: {
            priority: 2,
            className: 'heading class name'
        }
    }}
    onChange={handleChange}
/>
```
