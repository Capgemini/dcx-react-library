### An example of 'un-styled' Input

<Canvas>
  <Story name="un-styled">
    {(args, { viewMode }) => {
      if (viewMode === 'docs') {
        const [value, setValue] = React.useState('');
        const handleChange = event => {
          setValue(event.currentTarget.value);
        };
        return (
          <FormRadio
            label="text example"
            name="group1"
            value="value"
            id="radio-1"
            onChange={handleChange}
          />
        );
      } else {
        return <FormRadioLive />;
      }
    }}
    />
  </Story>
</Canvas>
