# Select

FormSelect component ready to use in your UI  
FormSelect is UI/UX agnostic so you need to provide your style to have the look and feel you prefer.  
When you import the button component without providing any className or style associated will looks as following:

<Canvas of={FormSelectStories.Unstyled} />
## Usage

Use the FormSelect component when users need to select a value form a list  
An example with all the available properties is:

```js
  <FormSelect
    nullOption="Select..."
    containerProps={containerProps}
    disabled={false}
    selectProps={{}}
    className="select classes"
    containerErrorClassName="containerErrorClassName"
    containerFilledClassName="containerFilledClassName"
    variant="floating"
    style={{
      fontSize: '10px'
    }}
    options={[
      { label: 'option1', value: 'value1' },
      { label: 'option2', value: 'value2' },
    ]}
    optionGroups={[
      {
        label: 'group 1',
        displayCount: true,
        options: [
          { label: 'option 1', value: 'group 1 option 1' },
          { label: 'option 2', value: 'group 1 option 2' },

          { label: 'option 4', value: 'group 1 option 4' },
          { label: 'option 5', value: 'group 1 option 5' },
          {
            label: 'option 6',
            value: 'group 1 option 6',
            disabled: true,
          },
        ],
      },
      {
        label: 'group 2',
        displayCount: true,
        options: [
          { label: 'option 1', value: 'group 2 option 1' },
          { label: 'option 2', value: 'group 2 option 2' },
          { label: 'option 3', value: 'group 2 option 3' },
        ],
      },
      {
        label: 'group 3',
        displayCount: true,
        options: [
          { label: 'option 1', value: 'group 3 option 1' },
          { label: 'option 2', value: 'group 3 option 2' },
          { label: 'option 3', value: 'group 3 option 3' },
          { label: 'option 4', value: 'group 3 option 4' },

          {
            label: 'option 6',

            disabled: true,

        ],
      },
      {
        label: 'group 4',

        options: [
          { label: 'option 1', value: 'group 4 option 1' },
          { label: 'option 2', value: 'group 4 option 2' },
          { label: 'option 3', value: 'group 4 option 3' },
          {
            label: 'option 4',
            value: 'group 4 option 4',

          },
        ],
      },
      {
        label: 'group 5',
        displayCount: true,
        options: [
          { label: 'option 1', value: 'group 4 option 1' },

      className: "select options classes"
    }}
    ariaLabel="select aria label"

      className: "label classes"

      text: "hint text",

      id: "hint id"
    }}
    error={{
      text: "error text",
      className: "error classes",
      id: "error id"
      visuallyHiddenText: {
        text: "visually hidden text",
        className: "visually hidden classes"
      }
    }}
    defaultValue="option1"
    tabIndex={0}
  />
```
