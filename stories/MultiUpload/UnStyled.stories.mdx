### An example of 'un-styled' MultiUpload component

<Canvas>
  <Story name="un-styled">
    {() => {
      const handleChange = file => alert(`${file.name} was uploaded, it was last modified at ${new Date(file.lastModified).toLocaleDateString("en-us")}`);
      return (<MultiUpload name="file-upload" label="Upload File" onChange={handleChange}/>);
    }}
  </Story>
</Canvas>

### Multi upload with hint

<Canvas>
  <Story name="hint">
    {() => {
      const handleChange = file => alert(`${file.name} was uploaded, it was last modified at ${new Date(file.lastModified).toLocaleDateString("en-us")}`);
      return (
        <MultiUpload
          name="file-upload-hint"
          acceptedFormats=".jpg, .png"
          label="Upload File"
          hint={{
            text: 'This can be in either JPG or PNG format',
            className: 'classes for hint',
            id: 'hint-id',
          }}
          onChange={handleChange}
        />
      );
    }}
  </Story>
</Canvas>
