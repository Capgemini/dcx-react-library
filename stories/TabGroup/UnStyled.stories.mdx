            </Tab>

            </Tab>
            <Tab label="Tab 4" eventKey="tab-pane-4-id">

            </Tab>
            <Tab label="Tab 5" eventKey="tab-pane-5-id">
              This is content for <em>tab 5</em>
            </Tab>
            </TabGroup>
            <Button onClick={() => {
              //@ts-ignore
              const changed = tabRef.current.updateActiveTab('tab-pane-5-id');
            }} label="Click to change to Tab 5" />
        </>
      )
    }}
  </Story>
</Canvas>
