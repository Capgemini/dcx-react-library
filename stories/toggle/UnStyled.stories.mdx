### Examples of basic Toggle

<Canvas>
  <Story name="basic">
    {() => {
      const [checked, setChecked] = React.useState(false);
      return (
        <Toggle
          checked={checked}
          onChange={e => {
            setChecked(e);
          }}
          onColor="green"
          offColor="gray"
        />
      );
    }}
  </Story>
</Canvas>
