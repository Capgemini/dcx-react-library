### Examples of basic Select

<Canvas>
  <Story name="basic">
    {() => {
      const [value, setValue] = React.useState('');
      const handleChange = evt => {

      return (
        <FormSelect
          id="myId"
          name="the name"
          value={value}

            { label: 'option1', value: 'value1' },
            { label: 'option2', value: 'value2' },
          ]}
        />
      );
    }}
  </Story>
</Canvas>
