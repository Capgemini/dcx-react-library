### An example of 'un-styled' CheckboxGroup of checkbox buttons

<Canvas>
  <Story name="un-styled-checkboxes">
    {() => {
      const [selected, setSelected] = React.useState('');
      const handleChange = event => setSelected(event.currentTarget.value);
      return (
        <CheckboxGroup
          name="name-of-group"
          type="checkbox"
          legend={{
            text:
              'Heading text customisable by a given priority of type number between 1 to 6 (inclusive)',
            heading: {
              priority: 2,
            },
          }}
          items={[
            {
              label: 'item-label',
              value: 'yes',
              id: 'first',
            },
            {
              label: 'item-label',
              value: 'no',
              id: 'second',
            },
          ]}
          onChange={handleChange}
        />
      );
    }}
    />
  </Story>
</Canvas>
