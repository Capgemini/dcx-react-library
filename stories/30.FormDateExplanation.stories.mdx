# Form Date

A form date component ready to use in your form.

FormDate is UI/UX agnostic so you need to provide your style to have the look and feel you prefer.

## Usage

Use the form date component when you want to enforce the correct format of a date.

It provides the ability to specify your date format. The allowed formats are:

- `dd/mm/yyyy`
- `dd/mm/yy`
- `yyyy/mm/dd`
- `yy/mm/dd`

An example with all the available properties is:

```js
const [isValid, setIsValid] = React.useState(false);
const [date, setDate] = React.useState(0);
const [showError, setShowError] = React.useState(false);
const handleValidity = (valid: boolean, date: number) => {
  valid === false ? setShowError(true) : setShowError(false);
  setIsValid(valid);
  setDate(date);
};
return (
  <>
    <FormDate
      dateFormat="dd/mm/yyyy"
      inputContainerClass={showError ? 'errorContainer' : ''}
      displayError={showError}
      errorMessage={showError ? 'Enter a valid date' : null}
      errorPosition="top"
      yearProps={{
        classNameLabel: 'yearLabel',
        customLabel: <CustomComponent>Year</CustomComponent>, //you can specify your customLabel component
        classNameLabel: 'yearLabel',
        classNameSpan: 'span',
      }}
      monthProps={{
        label: 'Month',
        classNameInput: 'classInput',
        classNameLabel: 'monthLabel',
        classNameSpan: 'classSpan',
      }}
      dayProps={{
        classNameLabel: 'dayLabel',
        label: 'Day',
        classNameInput: 'classInput',
        classNameLabel: 'classDayLabel',
        classNameSpan: 'classSpan',
      }}
      day="29"
      month="07"
      year="1982"
      disabled={true}
    />
    <pre>isValid: {isValid.toString()}</pre>
    <pre>date: {JSON.stringify(new Date(date))}</pre>
  </>
);
```
